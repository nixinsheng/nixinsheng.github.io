### iOS - 计算某一年第几周的起始日和结束日
##### 需求
遇到这样一个需求: 获取某一年的第几周的第一天和最后一天的日期(也就是这一周的周日和周六的日期), 比如需求是获取2017年的第20周的起始日和结束日的日期, 在网上搜索了一遍后都没发现相关的代码, 于是我就自己写了个方法来实现这个需求, 代码只有下面这么点, 有需要的朋友拷贝进项目调用就好.

/**
 获取指定年限第n周的周一的日期(精确到天)

 @param year 指定年
 @param weeks 第n周
 @return 指定年限第n周的周一的日期(精确到天)
 */
-(NSString*)getStartDateInYear:(NSUInteger)year andWeeks:(NSUInteger)weeks{
    NSDate *startDate;
    NSDate *endDate;
    [self calculateStartDate:&startDate endDate:&endDate inYear:year andWeeks:weeks];
    NSDateFormatter *outputFormatter = [[NSDateFormatter alloc] init];
    [outputFormatter setLocale:[NSLocale currentLocale]];
    [outputFormatter setDateFormat:@"yyyy-MM-dd"];
    NSString *strStart = [outputFormatter stringFromDate:[startDate dateByAddingDays:1]];
    NSString *strEnd = [outputFormatter stringFromDate:[endDate dateByAddingDays:1]];
    NSLog(@"===[%@,%@]===",strStart,strEnd);
    startDate = nil;
    endDate = nil;
    return [NSString stringWithFormat:@"%@",strStart];
}

 参考资料地址：https://www.aliyun.com/jiaocheng/360437.html(待抽取优化)

- (void)calculateStartDate:(out NSDate * __autoreleasing*)startDate endDate:(out NSDate * __autoreleasing*)endDate inYear:(NSUInteger)year andWeeks:(NSUInteger)weeks{
    if (year == 0 || weeks == 0) {
        return;
    }
    NSDateFormatter *formatter = [[NSDateFormatter alloc] init];
    formatter.dateStyle = NSDateFormatterMediumStyle;
    formatter.timeStyle = NSDateFormatterShortStyle;
    formatter.dateFormat = @"yyyy-MM-dd";
    formatter.timeZone = [NSTimeZone timeZoneWithName:@"Asia/Beijing"];
    NSUInteger startMonth = 1;
    NSUInteger startDay = 1;
    NSUInteger endMonth = 1;
    NSUInteger endDay = 1;
    // 这一年的总天数
    NSUInteger totalDays = [self numberOfDaysInYear:year];
    // 这一年的第一天是星期几(一周的开始日为周日)
    NSDateComponents *firstDayComponent = [[NSCalendar currentCalendar] components:NSCalendarUnitWeekday fromDate:[self firstDateOfYear:year]];
    NSUInteger weekday = firstDayComponent.weekday;
    // 第一周
    if (weeks == 1) {
        startMonth = 1;
        startDay = 1;
        endMonth = 1;
        endDay = 7 - (weekday - 1);
        if (startDate) {
            *startDate = [formatter dateFromString:[NSString stringWithFormat:@"%zi-%zi-%zi", year, startMonth, startDay]];
        }
        if (endDate) {
            *endDate = [formatter dateFromString:[NSString stringWithFormat:@"%zi-%zi-%zi", year, endMonth, endDay]];
        }
        return;
    }
    // 计算输入的周数是否满足这一年的最大周数
    NSUInteger totalWeeks = totalDays/7;
    if (totalDays % 7 > 0) {
        totalWeeks += 1;
    }
    if (weeks > totalWeeks) {
        weeks = totalWeeks;
    }
    // 这一年从第一天到指定周的结束日的总天数
    NSUInteger targetEndDays = weeks*7 - (weekday - 1);
    if (targetEndDays > totalDays) { // 最后一周不满一周, 则直接设置结束的总天数为这一年的总天数
        targetEndDays = totalDays;
    }
    // 这一年从第一天到指定周的开始日的总天数
    NSUInteger targetStartDays = weeks*7 - (weekday - 1) - 6;
    // 是否获取到了开始日的日期
    BOOL getStart = NO;
    // 是否获取到了结束日的日期
    BOOL getEnd = NO;
    // 累加遍历了的总天数
    NSUInteger cumsumDays = 0;
    // 遍历这一年
    for (int month = 1; month <= 12; month ++) {
        NSDate *date = [formatter dateFromString:[NSString stringWithFormat:@"%zi-%d-01", year, month]];
        // 这个月的总天数
        NSUInteger numberOfDays = [[NSCalendar currentCalendar] rangeOfUnit:NSCalendarUnitDay inUnit:NSCalendarUnitMonth forDate:date].length;
        // 遍历这一个月
        for (int day = 1; day <= numberOfDays; day ++) {
            cumsumDays ++;
            if (cumsumDays == targetStartDays) {
                startMonth = month;
                startDay = day;
                getStart = YES;
            }
            if (cumsumDays == targetEndDays) {
                endMonth = month;
                endDay = day;
                getEnd = YES;
                break;
            }
        }
        if (getStart && getEnd) {
            break;
        }
    }
    if (startDate) {
        *startDate = [formatter dateFromString:[NSString stringWithFormat:@"%zi-%zi-%zi", year, startMonth, startDay]];
    }
    if (endDate) {
        *endDate = [formatter dateFromString:[NSString stringWithFormat:@"%zi-%zi-%zi", year, endMonth, endDay]];
    }
}
- (NSUInteger)numberOfDaysInYear:(NSInteger)year
{
    NSDate *firstDateOfYear = [self firstDateOfYear:year];
    NSDate *firstDateOfNextYear = [self firstDateOfYear:year + 1];
    NSDateComponents *components = [[NSCalendar currentCalendar] components:NSCalendarUnitDay fromDate:firstDateOfYear toDate:firstDateOfNextYear options:0];
    return [components day];
}
- (NSDate *)firstDateOfYear:(NSInteger)year{
    NSDateComponents *dc = [[NSDateComponents alloc] init];
    dc.year = year;
    dc.month = 1;
    dc.day = 1;
    return [[NSCalendar currentCalendar] dateFromComponents:dc];
}
