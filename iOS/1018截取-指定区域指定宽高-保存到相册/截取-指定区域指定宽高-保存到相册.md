### 2.保存截图到相册如下:
```
#pragma mark - 按钮监听
// 保存截图到相册
- (void)saveClick {
    DLog(@"保存截图到相册");
    //UIImageWriteToSavedPhotosAlbum([self UIViewToUIImageView:self.view], self,  @selector(image:didFinishSavingWithError:contextInfo:), NULL);
    //self.containerView
    UIImageWriteToSavedPhotosAlbum([self UIViewToUIImageView:self.containerView], self,  @selector(image:didFinishSavingWithError:contextInfo:), NULL);
}
- (UIImage*)UIViewToUIImageView:(UIView*)view{
    //也就是底部两个按钮的高度
    CGFloat Delt = 60.0;
    if (IS_IPHONE_5||IS_IPHONE_6_7) {
        Delt = 50.0;
    }
    CGSize size = CGSizeMake(self.containerView.frame.size.width,self.containerView.frame.size.height-Delt);
    // 下面的方法：第一个参数表示区域大小；第二个参数表示是否是非透明的如果需要显示半透明效果，需要传NO，否则传YES；第三个参数是屏幕密度
    UIGraphicsBeginImageContextWithOptions(size, YES, [UIScreen mainScreen].scale);
    [view.layer renderInContext:UIGraphicsGetCurrentContext()];
    UIImage*image = UIGraphicsGetImageFromCurrentImageContext();
    UIGraphicsEndImageContext();
    return image;
}
// 保存相册回调
- (void)image:(UIImage *)image didFinishSavingWithError:(NSError *)error contextInfo:(void *)contextInfo {
    NSString *msg = nil ;
    if(error != NULL){
        msg = @"保存图片失败" ;
    }else{
        msg = @"保存图片成功" ;
    }
    [self showAlertControllerWithText:msg];
}

```
